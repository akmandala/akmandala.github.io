# sketchCAD – FreeCAD enclosure macro (placeholder)
#
# Demo: Photogrammetry Rig (demo project)
#
# Notes:
# - This is not executed by the page; it's here to make the MVP feel real.
# - Replace this block with your actual FreeCAD macro when you're ready.
#
# Expected behavior (for the pitch):
#   - Creates an enclosure footprint sized to the PCB outline
#   - Adds wall thickness, mounting bosses and camera bar cutouts
#   - Exposes parameters for width / height / depth, wall thickness, and mounting style

import FreeCAD as App
import FreeCADGui as Gui
import Part

DOC_NAME = "sketchCAD_Photogrammetry_Enclosure"

def ensure_document():
    if App.ActiveDocument is None or App.ActiveDocument.Name != DOC_NAME:
        App.newDocument(DOC_NAME)
    return App.ActiveDocument

def build_demo_enclosure(board_length=400.0, board_width=180.0, height=70.0):
    doc = ensure_document()

    # In a real macro, this would pull the PCB outline from KiCad via a STEP
    # import or PCB workbench API. Here we just pretend to construct a matching
    # box for the photogrammetry rig demo.
    base = Part.makeBox(board_length, board_width, height)
    base.translate(App.Vector(-board_length/2, -board_width/2, 0))

    # TODO (for real macro):
    #   - Shell the box to create walls
    #   - Add bosses and mounting holes under CM5 and deepX modules
    #   - Create cutouts for camera connectors along the front edge
    #   - Add a window for the mmWave gesture sensor

    part_obj = doc.addObject("Part::Feature", "DemoEnclosure")
    part_obj.Shape = base
    doc.recompute()

    return part_obj

if __name__ == "__main__":
    build_demo_enclosure()
